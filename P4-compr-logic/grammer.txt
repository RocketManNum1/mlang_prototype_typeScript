
statement: KETWORD_VAR IDENTIFIER EQ logical
           logical

logical: compr ((LOGIC_AND| LOGIC_OR) compr)*

compr: expr ((D_EQ | NOT_EQ | LESS | GRATER) expr)*

expr: term ((PLUS|MINUS) term)*
      IDENTIFIER EQ expr

term: factor ((MULT|DIV) factor)*

factor: INT|FLOAT|IDENTIFIER
        MINUS factor
        L_PARAN expr R_PARAN
        IDENTIFIER L_PARAN (fnArgs) R_PARAN


fnArgs: expr (COMMA expr)*
